-- สคริปต์นี้จะสร้างตารางที่จำเป็นทั้งหมด ตั้งค่าความปลอดภัย และเพิ่มข้อมูลเริ่มต้นให้ครับ

-- ==========================================
-- 1. สร้างตารางพื้นฐาน (Table Definitions)
-- ==========================================

-- ตารางผู้ใช้งาน
CREATE TABLE IF NOT EXISTS public.users (
  id TEXT PRIMARY KEY,
  name TEXT,
  email TEXT UNIQUE,
  image TEXT,
  password TEXT,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- ตารางการตั้งค่าแอป
CREATE TABLE IF NOT EXISTS public.settings (
  user_id TEXT DEFAULT 'global',
  key TEXT, 
  value TEXT,
  PRIMARY KEY(user_id, key)
);

-- ตารางหมวดหมู่หลักสูตร
CREATE TABLE IF NOT EXISTS public.degree_categories (
  user_id TEXT DEFAULT 'global',
  id TEXT, 
  name TEXT NOT NULL, 
  required INTEGER NOT NULL,
  PRIMARY KEY(user_id, id)
);

-- ตารางรายวิชาในแต่ละหมวดหมู่
CREATE TABLE IF NOT EXISTS public.degree_courses (
  user_id TEXT DEFAULT 'global',
  category_id TEXT, 
  course_code TEXT, 
  PRIMARY KEY(user_id, category_id, course_code)
);

-- ตารางปฏิทินกิจกรรม (ใช้เครื่องหมายคำพูดเพื่อรักษาตัวพิมพ์ใหญ่-เล็ก)
CREATE TABLE IF NOT EXISTS public.events (
  id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  user_id TEXT DEFAULT 'global',
  title TEXT NOT NULL,
  "startDate" TEXT,
  "endDate" TEXT,
  type TEXT NOT NULL,
  region TEXT NOT NULL,
  "dateStr" TEXT,
  "sendNotify" BOOLEAN DEFAULT FALSE
);

-- ตารางวิชาที่เรียนจบแล้ว/สอบซ่อม
CREATE TABLE IF NOT EXISTS public.completed_courses (
  user_id TEXT DEFAULT 'global',
  course_code TEXT, 
  grade TEXT,
  is_reexam BOOLEAN DEFAULT FALSE,
  PRIMARY KEY(user_id, course_code)
);

-- ตารางฐานข้อมูลวิชาทั้งหมด (Master Data)
CREATE TABLE IF NOT EXISTS public.courses (
  code TEXT PRIMARY KEY,
  name TEXT NOT NULL,
  credit INTEGER NOT NULL,
  day TEXT,
  time TEXT,
  room TEXT,
  "examDate" TEXT,
  "examTime" TEXT
);

-- ตารางวิชาในตารางเรียนที่วางแผนไว้
CREATE TABLE IF NOT EXISTS public.planner_courses (
  user_id TEXT DEFAULT 'global',
  course_code TEXT,
  PRIMARY KEY(user_id, course_code)
);

-- ตารางแผนการเรียนในแต่ละเทอม (Roadmap)
CREATE TABLE IF NOT EXISTS public.semester_plans (
  user_id TEXT DEFAULT 'global',
  semester_id TEXT,
  course_code TEXT,
  PRIMARY KEY(user_id, semester_id, course_code)
);

-- ==========================================
-- 2. จัดการเรื่อง Case Sensitivity ของคอลัมน์
-- (สำหรับกรณีที่คุณมีตารางเดิมอยู่แล้วแต่ชื่อคอลัมน์เป็นตัวพิมพ์เล็กหมด)
-- ==========================================
DO $$ 
BEGIN
    -- สำหรับ events
    IF EXISTS (SELECT 1 FROM information_schema.columns WHERE table_name='events' AND column_name='startdate') THEN
        ALTER TABLE public.events RENAME COLUMN startdate TO "startDate";
    END IF;
    IF EXISTS (SELECT 1 FROM information_schema.columns WHERE table_name='events' AND column_name='enddate') THEN
        ALTER TABLE public.events RENAME COLUMN enddate TO "endDate";
    END IF;
    IF EXISTS (SELECT 1 FROM information_schema.columns WHERE table_name='events' AND column_name='datestr') THEN
        ALTER TABLE public.events RENAME COLUMN datestr TO "dateStr";
    END IF;
    IF EXISTS (SELECT 1 FROM information_schema.columns WHERE table_name='events' AND column_name='sendnotify') THEN
        ALTER TABLE public.events RENAME COLUMN sendnotify TO "sendNotify";
    END IF;

    -- สำหรับ courses
    IF EXISTS (SELECT 1 FROM information_schema.columns WHERE table_name='courses' AND column_name='examdate') THEN
        ALTER TABLE public.courses RENAME COLUMN examdate TO "examDate";
    END IF;
    IF EXISTS (SELECT 1 FROM information_schema.columns WHERE table_name='courses' AND column_name='examtime') THEN
        ALTER TABLE public.courses RENAME COLUMN examtime TO "examTime";
    END IF;
END $$;

-- ==========================================
-- 3. ตั้งค่าสิทธิ์การเข้าถึง (Row Level Security)
-- ==========================================

-- เปิดการใช้งาน RLS สำหรับทุกตาราง
ALTER TABLE public.users ENABLE ROW LEVEL SECURITY;
ALTER TABLE public.settings ENABLE ROW LEVEL SECURITY;
ALTER TABLE public.degree_categories ENABLE ROW LEVEL SECURITY;
ALTER TABLE public.degree_courses ENABLE ROW LEVEL SECURITY;
ALTER TABLE public.events ENABLE ROW LEVEL SECURITY;
ALTER TABLE public.completed_courses ENABLE ROW LEVEL SECURITY;
ALTER TABLE public.courses ENABLE ROW LEVEL SECURITY;
ALTER TABLE public.planner_courses ENABLE ROW LEVEL SECURITY;
ALTER TABLE public.semester_plans ENABLE ROW LEVEL SECURITY;

-- สร้างนโยบาย "Allow all access" แบบวนลูป (สำหรับการเริ่มต้นพัฒนา)
DO $$
DECLARE
    t text;
BEGIN
    FOR t IN SELECT tablename FROM pg_tables WHERE schemaname = 'public'
    LOOP
        IF NOT EXISTS (SELECT 1 FROM pg_policies WHERE policyname = 'Allow all access' AND tablename = t) THEN
            EXECUTE format('CREATE POLICY "Allow all access" ON public.%I FOR ALL USING (true)', t);
        END IF;
    END LOOP;
END $$;
